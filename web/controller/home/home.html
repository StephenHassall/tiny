What the code looks like
Load in HTML and CSS files dynamically. Set local variables. Handle the loaded event. Create template area. Process inline event.
1import Controller from "/lib/tiny/controller.js";
2import Template from "/lib/tiny/template.js";
3
4/**
5 * Luck 7
6 *
7 * This web component shows how the Template class is used to set HTML text areas using
8 * string templates.
9 */
10export default class Lucky7Controller extends Controller {
11  /**
12   * Lucky 7 controller web component constructor.
13   * @constructor
14   */
15  constructor() {
16    // Call the super with the page to load
17    super({
18      htmlPath: 'lucky7.html',
19      cssPath: 'lucky7.css',
20      importMetaUrl: import.meta.url,
21      useShadowDom: true
22    });
23
24    // Set the dice parts
25    this.dice1 = Math.floor(Math.random() * 6) + 1;
26    this.dice2 = Math.floor(Math.random() * 6) + 1;
27    this.diceTotal = this.dice1 + this.dice2;
28  }
29
30  /**
31   * Override loadCallback function to handle when HTML/CSS has been loaded.
32   * @override
33   */
34  loadedCallback() {
35    // Create template area object
36    this._templateArea = new Template('templateArea', this);
37
38    // Update the template area for the first time
39    this._templateArea.update();
40  }
41
42  /**
43   * Role dice button event.
44   */
45  roleDice() {
46    // Reset the dice parts
47    this.dice1 = Math.floor(Math.random() * 6) + 1;
48    this.dice2 = Math.floor(Math.random() * 6) + 1;
49    this.diceTotal = this.dice1 + this.dice2;
50
51    // Update the template area with the new values
52    this._templateArea.update();
53  }
54}
55
56// Define controller web component
57customElements.define('lucky7-controller', Lucky7Controller);
